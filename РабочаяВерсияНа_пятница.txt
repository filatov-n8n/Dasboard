<script>
  let SHOW_POINTS = false;
  
  function normalizeDeptName(dep) {
    const d = String(dep || '').toLowerCase().trim();
    if (d === 'доставка') return 'Delivery';
    if (d === 'установка') return 'Installation';
    if (d === 'орк') return 'Customer';
    if (d === 'call центр' || d === 'колл центр') return 'Call Center';
    return dep;
  }

  function normalizeMonth(m) {
    const map = {
      'январь': '01', 'февраль': '02', 'март': '03', 'апрель': '04', 'май': '05', 'июнь': '06',
      'июль': '07', 'август': '08', 'сентябрь': '09', 'октябрь': '10', 'ноябрь': '11', 'декабрь': '12'
    };
    const key = String(m || '').toLowerCase().trim();
    return map[key] || key;
  }

  function parseValue(v) {
    if (typeof v === 'string') {
      const s = v.replace(',', '.').replace('%', '').trim();
      const num = parseFloat(s);
      return isNaN(num) ? 0 : num;
    }
    return Number(v) || 0;
  }

  const APP = {
    raw: [],
    currentDept: 'Delivery',
    years: new Set(),
    months: new Set(),
    selectedYears: [],
    selectedMonths: new Set(),
    chartInstances: [],
    maxMinData: {}
  };

  const MONTH_ORDER = ['01', '02', '03', '04', '05', '06', '07', '08', '09', '10', '11', '12'];
  const MONTH_LABELS = {
    '01': 'Янв', '02': 'Фев', '03': 'Мар', '04': 'Апр', '05': 'Май', '06': 'Июн',
    '07': 'Июл', '08': 'Авг', '09': 'Сен', '10': 'Окт', '11': 'Ноя', '12': 'Дек'
  };
  const YEAR_COLOR_MAP = { 2025: '#3B82F6', 2026: '#F59E0B' };
  const YEAR_COLORS = ['#00B0FF', '#FFD54F', '#00E676', '#FF5252', '#67757B'];

  function getYearColor(year, index) {
    return YEAR_COLOR_MAP[year] || YEAR_COLORS[index % YEAR_COLORS.length];
  }

  function getAveragePercentValue(rows) {
    if (!rows || !rows.length) return 0;
    const values = rows.map(item => Number(item.value) || 0).filter(value => Number.isFinite(value));
    if (!values.length) return 0;
    return (values.reduce((acc, value) => acc + value, 0) / values.length) * 100;
  }

  function parseDataToApp(dataArray) {
    APP.years = new Set();
    APP.months = new Set();
    APP.raw = (dataArray || []).map(r => {
      const dep = normalizeDeptName(r['Department']);
      const year = Number(r['Year']);
      const monthCode = normalizeMonth(r['Month']);
      const chart = Number(r['Chart #'] || r['Chart'] || r['Chart#']);
      const indicator = r['Indicator'];
      const value = parseValue(r['Value']);

      if (year) APP.years.add(year);
      if (monthCode) APP.months.add(monthCode);

      return {
        dept: dep,
        year,
        monthLabel: r['Month'],
        monthCode,
        chart,
        indicator,
        value,
        raw: r
      };
    });
  }

  function getAvailableYears() {
    return Array.from(APP.years).filter(Boolean).sort((a, b) => b - a);
  }

  function getActiveYears() {
    const availableYears = getAvailableYears();
    if (!availableYears.length) return [];

    if (APP.selectedYears && APP.selectedYears.length) {
      return availableYears.filter(year => APP.selectedYears.includes(year));
    }

    return [availableYears[0]];
  }

  function renderYearMonthControls() {
    const yearsArr = getAvailableYears();
    const monthsArr = Array.from(APP.months).filter(Boolean).sort((a, b) => MONTH_ORDER.indexOf(a) - MONTH_ORDER.indexOf(b));

    const yearsEl = document.getElementById('years-list');
    const monthsEl = document.getElementById('months-grid');
    if (!yearsEl || !monthsEl) return;

    yearsEl.innerHTML = '';
    monthsEl.innerHTML = '';

    yearsArr.forEach(year => {
      const btn = document.createElement('button');
      btn.className = 'year-pill w-full px-2 py-1 rounded-full text-[11px] border border-slate-700 text-slate-200 hover:bg-slate-700';
      btn.dataset.year = year;
      btn.textContent = year;
      yearsEl.appendChild(btn);
    });

    monthsArr.forEach(month => {
      const btn = document.createElement('button');
      btn.className = 'month-pill px-2 py-1 rounded-full border border-slate-700 text-slate-200 hover:bg-slate-700 text-[11px]';
      btn.dataset.month = month;
      btn.textContent = MONTH_LABELS[month] || month;
      monthsEl.appendChild(btn);
    });

    setupYearMonthSelection();
  }

  function setupYearMonthSelection() {
    const yearButtons = document.querySelectorAll('.year-pill');
    const monthButtons = document.querySelectorAll('.month-pill');

    const yearsArr = getAvailableYears();
    APP.selectedYears = yearsArr.length ? [yearsArr[0]] : [];
    APP.selectedMonths = new Set(APP.months);

    yearButtons.forEach(btn => {
      const year = Number(btn.dataset.year);
      const isActive = APP.selectedYears.includes(year);
      btn.classList.toggle('pill-active', isActive);
      btn.onclick = () => {
        const selectedYears = [...APP.selectedYears];
        const yearIndex = selectedYears.indexOf(year);

        if (yearIndex >= 0) {
          if (selectedYears.length > 1) {
            selectedYears.splice(yearIndex, 1);
          }
        } else if (selectedYears.length < 2) {
          selectedYears.push(year);
        } else {
          selectedYears.splice(0, 1);
          selectedYears.push(year);
        }

        APP.selectedYears = selectedYears;
        updateYearButtonStates();
        applyFiltersAndRender();
      };
    });

    monthButtons.forEach(btn => {
      const month = btn.dataset.month;
      const isActive = APP.selectedMonths.has(month);
      btn.classList.toggle('pill-active', isActive);
      btn.onclick = () => {
        const selectedMonths = new Set(APP.selectedMonths);
        if (selectedMonths.has(month) && selectedMonths.size > 1) {
          selectedMonths.delete(month);
        } else if (!selectedMonths.has(month)) {
          selectedMonths.add(month);
        }

        APP.selectedMonths = selectedMonths;
        updateMonthButtonStates();
        applyFiltersAndRender();
      };
    });

    updateYearButtonStates();
    updateMonthButtonStates();
  }

  function updateYearButtonStates() {
    document.querySelectorAll('.year-pill').forEach(btn => {
      const year = Number(btn.dataset.year);
      btn.classList.toggle('pill-active', APP.selectedYears.includes(year));
    });
  }

  function updateMonthButtonStates() {
    document.querySelectorAll('.month-pill').forEach(btn => {
      const month = btn.dataset.month;
      btn.classList.toggle('pill-active', APP.selectedMonths.has(month));
    });
  }

  function applyFiltersAndRender() {
    if (!APP.raw.length) return;

    const dept = APP.currentDept;
    const activeYears = getActiveYears();
    const years = new Set(activeYears);
    const months = APP.selectedMonths.size ? APP.selectedMonths : APP.months;

    const inSelection = r => years.has(r.year) && months.has(r.monthCode);

    const kpiRowsByDept = deptCode => APP.raw.filter(r => r.dept === deptCode && inSelection(r) && r.chart === 1);
    const kpiMap = {
      'Delivery': { valId: 'kpi-delivery-val', topId: 'kpi-delivery-top', minId: 'kpi-delivery-min' },
      'Installation': { valId: 'kpi-installation-val', topId: 'kpi-installation-top', minId: 'kpi-installation-min' },
      'Customer': { valId: 'kpi-customer-val', topId: 'kpi-customer-top', minId: 'kpi-customer-min' },
      'Call Center': { valId: 'kpi-call-center-val', topId: 'kpi-call-center-top', minId: 'kpi-call-center-min' }
    };

    Object.keys(kpiMap).forEach(code => {
      const rows = kpiRowsByDept(code);
      const sum = rows.reduce((acc, item) => acc + item.value, 0);
      const max = rows.reduce((acc, item) => Math.max(acc, item.value), 0) || 0;
      const min = rows.length > 0 ? rows.reduce((acc, item) => Math.min(acc, item.value), Infinity) : 0;
     
      document.getElementById(kpiMap[code].valId).textContent = sum ? sum.toLocaleString('ru-RU') : '0';
      document.getElementById(kpiMap[code].topId).textContent = max ? max.toLocaleString('ru-RU') : '0';
      document.getElementById(kpiMap[code].minId).textContent = (min !== Infinity) ? min.toLocaleString('ru-RU') : '0';
    });

    const rowsForDept = APP.raw.filter(r => r.dept === dept && inSelection(r));
    const byChart = {
      1: rowsForDept.filter(r => r.chart === 1),
      2: rowsForDept.filter(r => r.chart === 2),
      3: rowsForDept.filter(r => r.chart === 3)
    };

    document.getElementById('total-chart1').textContent = byChart[1].reduce((acc, item) => acc + item.value, 0).toLocaleString('ru-RU');
    document.getElementById('total-chart2').textContent = byChart[2].reduce((acc, item) => acc + item.value, 0).toLocaleString('ru-RU');
    const chart3Average = getAveragePercentValue(byChart[3]).toFixed(1);
    document.getElementById('total-chart3').textContent = chart3Average + '%';

    updateChartTitles(byChart);
    renderCharts(byChart);
  }

  function updateChartTitles(byChart) {
    const chart1Row = byChart[1] && byChart[1][0];
    const chart1Title = chart1Row ? chart1Row.indicator : 'Chart #1';
    document.getElementById('chart1-title').textContent = `Chart #1 · ${chart1Title || 'Indicator'}`;

    const chart2Row = byChart[2] && byChart[2][0];
    const chart2Title = chart2Row ? chart2Row.indicator : 'Chart #2';
    document.getElementById('chart2-title').textContent = `Chart #2 · ${chart2Title || 'Столбчатый по годам'}`;

    const chart3Row = byChart[3] && byChart[3][0];
    const chart3Title = chart3Row ? chart3Row.indicator : 'Chart #3';
    document.getElementById('chart3-title').textContent = `Chart #3 · ${chart3Title || 'Доли по годам'}`;
  }

  function getMonthsUsed(rows) {
    const monthCodes = [...new Set(rows.map(r => r.monthCode))].filter(Boolean);
    return monthCodes.sort((a, b) => MONTH_ORDER.indexOf(a) - MONTH_ORDER.indexOf(b));
  }

  function disposeChartInstances() {
    if (window._echartsInstances && window._echartsInstances.length) {
      window._echartsInstances.forEach(inst => {
        if (inst && typeof inst.dispose === 'function') inst.dispose();
      });
    }
    window._echartsInstances = [];
    APP.chartInstances = [];
  }

  function renderCharts(byChart) {
    disposeChartInstances();

    const rows1 = byChart[1] || [];
    const rows2 = byChart[2] || [];
    const rows3 = byChart[3] || [];

    const monthsUsed = getMonthsUsed([...rows1, ...rows2, ...rows3]);
    const xLabels = monthsUsed.map(month => MONTH_LABELS[month] || month);

    const selectedYears = getActiveYears().sort((a, b) => a - b);
    const chart1 = echarts.init(document.getElementById('chart1'));
    const chart2 = echarts.init(document.getElementById('chart2'));
    APP.chartInstances = [chart1, chart2];

    // Вычисляем макс/мин значения для подсвечивания
    APP.maxMinData = {};
    selectedYears.forEach(year => {
      const yearRows = rows1.filter(r => r.year === year);
      if (yearRows.length) {
        const maxVal = Math.max(...yearRows.map(r => r.value));
        const minVal = Math.min(...yearRows.map(r => r.value));
        APP.maxMinData[year] = { max: maxVal, min: minVal };
      }
    });

    renderChart1(chart1, rows1, monthsUsed, xLabels, selectedYears);
    renderChart2(chart2, rows2, monthsUsed, xLabels, selectedYears);
    renderChart3(rows3, selectedYears);
    updateLegends(selectedYears);

    window._echartsInstances = APP.chartInstances;
  }

  function renderDifferenceTooltip(activeYears, firstValue, secondValue) {
    if (!activeYears || activeYears.length !== 2) return '';

    const first = Number(firstValue);
    const second = Number(secondValue);
    if (!isFinite(first) || !isFinite(second) || first === 0) return '';

    const diff = ((second - first) / first) * 100;
    const color = diff >= 0 ? '#22c55e' : '#f59e0b';
    return '<div style="color:' + color + ';">' + (diff >= 0 ? '+' : '') + diff.toFixed(1) + '% (' + activeYears[0] + ' → ' + activeYears[1] + ')</div>';
  }

  function renderChart3(rows, selectedYears) {
    const container = document.getElementById('chart3');
    if (!container) return;
    const years = selectedYears.length ? selectedYears : Array.from(APP.years).sort((a, b) => a - b);
    container.innerHTML = '';
    if (!rows.length || !years.length) {
      container.innerHTML = '<div class="text-xs text-slate-400">Нет данных для Chart #3</div>';
      return;
    }

    const renderPieChart = (title, chartId, averageValue) => {
      const tile = document.createElement('div');
      tile.className = 'chart3-tile';
      tile.innerHTML = `
        <div style="display:flex;justify-content:space-between;align-items:center;margin-bottom:4px;">
          <span class="text-[11px] text-slate-300">${title}</span>
          <span style="background:rgba(59,130,246,.20);color:#FFFFFF;border:1px solid rgba(59,130,246,.35);padding:2px 8px;border-radius:999px;font-size:10px;font-weight:600;">
            Средний ${Number(averageValue || 0).toFixed(1)}%
          </span>
        </div>
        <div style="display:flex;justify-content:center;align-items:center;height:205px;">
          <div id="${chartId}" style="width:220px;height:220px;transform:translateY(-8px);"></div>
        </div>
      `;
      container.appendChild(tile);
      const pieDom = document.getElementById(chartId);
      if (!pieDom) return null;
      const oldPie = echarts.getInstanceByDom(pieDom);
      if (oldPie) oldPie.dispose();
      return echarts.init(pieDom);
    };

    const buildAverageData = (year, averageValue) => {
      const percent = Math.max(0, Math.min(100, Number(averageValue) || 0));
      return [
        {
          name: 'Средний %',
          value: percent,
          itemStyle: {
            color: getYearColor(year)
          }
        },
        {
          name: 'Остальное',
          value: 100 - percent,
          itemStyle: {
            color: '#1F2937'
          }
        }
      ];
    };

    const getAveragePercent = year => {
      const yearRows = rows.filter(r => r.year === year);
      return getAveragePercentValue(yearRows);
    };

    if (years.length === 1) {
      const year = years[0];
      const yearColor = getYearColor(year, 0);
      const pieChart = renderPieChart(String(year), 'pie-single', getAveragePercent(year));
      if (!pieChart) return;
      const data = buildAverageData(year, getAveragePercent(year));
      pieChart.setOption({
        color: [yearColor, '#334155'],
        animation: true,
        animationDuration: 800,
        tooltip: {
          trigger: 'item',
          formatter: function (params) {
            const averageValue = getAveragePercent(year);
            const label = params.name || '';
            const value = Number(params.value || 0);
            return '<div style="font-weight:600">' + label + '</div><div>Значение: ' + value.toFixed(1) + '%</div><div>Средний % за период: ' + averageValue.toFixed(1) + '%</div>';
          }
        },
        series: [{
          name: String(year),
          type: 'pie',
          radius: ['48%', '78%'],
          avoidLabelOverlap: false,
          label: { show: false },
          itemStyle: { borderRadius: 6, borderColor: '#020617', borderWidth: 2 },
          data
        }]
      });
      APP.chartInstances.push(pieChart);
      return;
    }

    const outerYear = years[0];
    const innerYear = years[1];
    const outerColor = getYearColor(outerYear, 0);
    const innerColor = getYearColor(innerYear, 1);
    const pieChart = renderPieChart(outerYear + ' vs ' + innerYear, 'pie-combined', (getAveragePercent(outerYear) + getAveragePercent(innerYear)) / 2);
    if (!pieChart) return;
    const outerData = buildAverageData(outerYear, getAveragePercent(outerYear));
    const innerData = buildAverageData(innerYear, getAveragePercent(innerYear));
    pieChart.setOption({
      animation: true,
      animationDuration: 800,
      tooltip: {
        trigger: 'item',
        formatter: function (params) {
          const label = params.name || '';
          const value = Number(params.value || 0);
          const yearLabel = params.seriesName || '';
          const yearRows = rows.filter(r => r.year === Number(yearLabel));
          const averageValue = getAveragePercentValue(yearRows);
          let html = '<div style="font-weight:600">' + label + ' (' + yearLabel + ')</div><div>Значение: ' + value.toFixed(1) + '%</div><div>Средний % за период: ' + averageValue.toFixed(1) + '%</div>';
          return html;
        }
      },
      series: [
        {
          name: String(outerYear),
          type: 'pie',
          radius: ['52%', '84%'],
          avoidLabelOverlap: false,
          label: { show: false },
          itemStyle: { borderRadius: 6, borderColor: '#020617', borderWidth: 2 },
          data: outerData
        },
        {
          name: String(innerYear),
          type: 'pie',
          radius: ['28%', '48%'],
          avoidLabelOverlap: false,
          label: { show: false },
          itemStyle: { borderRadius: 6, borderColor: '#020617', borderWidth: 2 },
          data: innerData
        }
      ]
    });
    APP.chartInstances.push(pieChart);
  }

  }

  function setupThemeToggle() {
    const body = document.body;
    const toggleBtn = document.getElementById('theme-toggle');
    if (!toggleBtn) return;
    const saved = localStorage.getItem('dashboard-theme');
    if (saved === 'light') body.classList.add('light-mode');
    toggleBtn.addEventListener('click', () => {
      body.classList.toggle('light-mode');
      const theme = body.classList.contains('light-mode') ? 'light' : 'dark';
      localStorage.setItem('dashboard-theme', theme);
      if (window._echartsInstances) {
        window._echartsInstances.forEach(inst => inst.resize());
      }
    });
  }
